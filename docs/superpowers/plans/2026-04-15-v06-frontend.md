-- 028: Request comments (broadcast + individual Q&A) + unread tracking

CREATE TABLE request_comment (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id       UUID NOT NULL REFERENCES tenant(id),
  request_id      UUID NOT NULL REFERENCES request(id) ON DELETE CASCADE,
  assignment_id   UUID REFERENCES assignment(id),
  author_user_id  UUID NOT NULL REFERENCES users(id),
  body            TEXT NOT NULL,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE INDEX request_comment_request_idx
  ON request_comment (request_id, created_at);
CREATE INDEX request_comment_assignment_idx
  ON request_comment (assignment_id, created_at)
  WHERE assignment_id IS NOT NULL;

-- RLS
ALTER TABLE request_comment ENABLE ROW LEVEL SECURITY;
CREATE POLICY request_comment_tenant_isolation ON request_comment
  USING (tenant_id = NULLIF(current_setting('app.tenant_id', true), '')::uuid);

-- Unread tracking: last time the assignee viewed the request detail
ALTER TABLE assignment ADD COLUMN last_viewed_at TIMESTAMPTZ;
```

// src/domain/comment/create.ts
import type pg from 'pg';
import { withTenant } from '../../db/with-tenant.js';
import type { ActorContext } from '../types.js';

export class CommentError extends Error {
  constructor(msg: string, readonly code: 'permission_denied' | 'not_found' | 'validation') {
    super(msg);
    this.name = 'CommentError';
  }
}

export type CreateCommentInput = {
  requestId: string;
  assignmentId: string | null;
  body: string;
};

export type CommentResult = { id: string; createdAt: string };

export async function createComment(
  pool: pg.Pool,
  actor: ActorContext,
  input: CreateCommentInput,
): Promise<CommentResult> {
  if (!input.body.trim()) {
    throw new CommentError('body required', 'validation');
  }

  return withTenant(pool, actor.tenantId, async (client) => {
    // Check request exists
    const { rows: reqRows } = await client.query(
      `SELECT created_by_user_id FROM request WHERE id = $1`,
      [input.requestId],
    );
    if (reqRows.length === 0) {
      throw new CommentError('request not found', 'not_found');
    }
    const isRequester = reqRows[0].created_by_user_id === actor.userId;

    if (input.assignmentId === null) {
      // Broadcast: only requester can post
      if (!isRequester) {
        throw new CommentError('only requester can broadcast', 'permission_denied');
      }
    } else {
      // Individual: must be the assignee or the requester
      const { rows: asg } = await client.query(
        `SELECT user_id FROM assignment WHERE id = $1 AND request_id = $2`,
        [input.assignmentId, input.requestId],
      );
      if (asg.length === 0) {
        throw new CommentError('assignment not found', 'not_found');
      }
      if (asg[0].user_id !== actor.userId && !isRequester) {
        throw new CommentError('not authorized for this thread', 'permission_denied');
      }
    }

    const { rows } = await client.query<{ id: string; created_at: Date }>(
      `INSERT INTO request_comment (tenant_id, request_id, assignment_id, author_user_id, body)
       VALUES ($1, $2, $3, $4, $5)
       RETURNING id, created_at`,
      [actor.tenantId, input.requestId, input.assignmentId, actor.userId, input.body],
    );

    return { id: rows[0].id, createdAt: rows[0].created_at.toISOString() };
  });
}
```

// src/domain/comment/list.ts
import type pg from 'pg';
import { withTenant } from '../../db/with-tenant.js';
import type { ActorContext } from '../types.js';

export type CommentItem = {
  id: string;
  authorUserId: string;
  authorName: string;
  body: string;
  createdAt: string;
};

export type ListCommentsResult = {
  broadcasts: CommentItem[];
  myThread: CommentItem[];
  allThreads?: Record<string, CommentItem[]>;
};

export async function listComments(
  pool: pg.Pool,
  actor: ActorContext,
  requestId: string,
): Promise<ListCommentsResult> {
  return withTenant(pool, actor.tenantId, async (client) => {
    // Check if actor is the requester
    const { rows: reqRows } = await client.query(
      `SELECT created_by_user_id FROM request WHERE id = $1`,
      [requestId],
    );
    const isRequester = reqRows.length > 0 && reqRows[0].created_by_user_id === actor.userId;

    // Broadcasts (assignment_id IS NULL)
    const { rows: bcRows } = await client.query(
      `SELECT rc.id, rc.author_user_id, u.display_name AS author_name, rc.body, rc.created_at
         FROM request_comment rc
         JOIN users u ON u.id = rc.author_user_id
        WHERE rc.request_id = $1 AND rc.assignment_id IS NULL
        ORDER BY rc.created_at ASC`,
      [requestId],
    );
    const broadcasts: CommentItem[] = bcRows.map(toItem);

    // My assignment's thread
    const { rows: myAsg } = await client.query(
      `SELECT id FROM assignment WHERE request_id = $1 AND user_id = $2`,
      [requestId, actor.userId],
    );
    let myThread: CommentItem[] = [];
    if (myAsg.length > 0) {
      const { rows: threadRows } = await client.query(
        `SELECT rc.id, rc.author_user_id, u.display_name AS author_name, rc.body, rc.created_at
           FROM request_comment rc
           JOIN users u ON u.id = rc.author_user_id
          WHERE rc.assignment_id = $1
          ORDER BY rc.created_at ASC`,
        [myAsg[0].id],
      );
      myThread = threadRows.map(toItem);
    }

    // Requester: all individual threads
    let allThreads: Record<string, CommentItem[]> | undefined;
    if (isRequester) {
      const { rows: allRows } = await client.query(
        `SELECT rc.id, rc.assignment_id, rc.author_user_id,
                u.display_name AS author_name, rc.body, rc.created_at
           FROM request_comment rc
           JOIN users u ON u.id = rc.author_user_id
          WHERE rc.request_id = $1 AND rc.assignment_id IS NOT NULL
          ORDER BY rc.assignment_id, rc.created_at ASC`,
        [requestId],
      );
      allThreads = {};
      for (const row of allRows) {
        const key = row.assignment_id;
        if (!allThreads[key]) allThreads[key] = [];
        allThreads[key].push(toItem(row));

// src/domain/org/tree.ts
import type pg from 'pg';
import { withTenant } from '../../db/with-tenant.js';
import type { ActorContext } from '../types.js';
import {
  canTargetOutsideScope,
  getVisibleOrgUnitIds,
} from '../request/permissions.js';

export type OrgTreeNode = {
  id: string;
  name: string;
  memberCount: number;
  children: OrgTreeNode[];
};

export async function getOrgTree(
  pool: pg.Pool,
  actor: ActorContext,
): Promise<OrgTreeNode[]> {
  return withTenant(pool, actor.tenantId, async (client) => {
    // Get visible org IDs
    let visibleIds: string[];
    if (canTargetOutsideScope(actor)) {
      const { rows } = await client.query<{ id: string }>(
        `SELECT id FROM org_unit`,
      );
      visibleIds = rows.map((r) => r.id);
    } else {
      visibleIds = await getVisibleOrgUnitIds(client, actor.userId);
    }

    if (visibleIds.length === 0) return [];

    // Fetch org_unit details + member counts
    const { rows: orgs } = await client.query<{
      id: string; name: string; parent_id: string | null; member_count: number;
    }>(
      `SELECT o.id, o.name, o.parent_id,
              (SELECT COUNT(*)::int FROM user_org_unit uou
                JOIN users u ON u.id = uou.user_id AND u.status = 'active'
               WHERE uou.org_unit_id = o.id) AS member_count
         FROM org_unit o
        WHERE o.id = ANY($1::uuid[])
        ORDER BY o.name`,
      [visibleIds],
    );

    // Build tree, excluding the tenant root (nodes whose parent_id IS NULL
    // or whose parent is not in the visible set)
    const byId = new Map(orgs.map((o) => [o.id, o]));
    const childrenMap = new Map<string | null, typeof orgs>();
    for (const o of orgs) {
      const parentKey = o.parent_id && byId.has(o.parent_id) ? o.parent_id : null;
      if (!childrenMap.has(parentKey)) childrenMap.set(parentKey, []);
      childrenMap.get(parentKey)!.push(o);
    }

    function buildNodes(parentId: string | null): OrgTreeNode[] {
      const children = childrenMap.get(parentId) ?? [];
      return children.map((o) => ({
        id: o.id,
        name: o.name,
        memberCount: o.member_count,
        children: buildNodes(o.id),
      }));
    }

    return buildNodes(null);
  });
}

  pool: pg.Pool,

  assignmentId: string,

// app/t/[code]/requests/page.tsx
import { cookies } from 'next/headers';
import { unsealSession } from '@/auth/session';
import { loadConfig } from '@/config';
import { appPool } from '@/db/pools';
import { withTenant } from '@/db/with-tenant';
import { StatusBadge } from '@/ui/components/status-badge';
import Link from 'next/link';

export const runtime = 'nodejs';

export default async function RequestListPage({
  params,
  searchParams,
}: {
  params: Promise<{ code: string }>;
  searchParams: Promise<{ status?: string; page?: string }>;
}) {
  const { code } = await params;
  const { status: statusFilter = 'pending', page: pageStr = '1' } = await searchParams;

  const cfg = loadConfig();
  const sealed = (await cookies()).get('nudge_session')?.value;
  const session = await unsealSession(sealed, cfg.IRON_SESSION_PASSWORD);
  if (!session) return <div>Unauthorized</div>;

  const page = Math.max(1, Number(pageStr) || 1);
  const pageSize = 20;

  const statusSql =
    statusFilter === 'done'
      ? `a.status IN ('responded','unavailable','forwarded','substituted','exempted','expired')`
      : `a.status IN ('unopened','opened')`;

  const data = await withTenant(appPool(), session.tenantId, async (client) => {
    const { rows: countRows } = await client.query<{ n: number }>(
      `SELECT COUNT(*)::int AS n FROM assignment a
         JOIN request r ON r.id = a.request_id
        WHERE a.user_id = $1 AND ${statusSql}`,
      [session.userId],
    );
    const { rows } = await client.query(
      `SELECT a.id, a.status, a.last_viewed_at,
              r.id AS request_id, r.title, r.due_at, r.created_at,
              u.display_name AS sender_name,
              (r.due_at IS NOT NULL AND r.due_at < now()
               AND a.status IN ('unopened','opened')) AS is_overdue,
              (SELECT MAX(rc.created_at) FROM request_comment rc
                WHERE rc.request_id = r.id
                  AND (rc.assignment_id IS NULL OR rc.assignment_id = a.id)
              ) AS last_comment_at
         FROM assignment a
         JOIN request r ON r.id = a.request_id
         JOIN users u ON u.id = r.created_by_user_id
        WHERE a.user_id = $1 AND ${statusSql}
        ORDER BY r.due_at ASC NULLS LAST, a.created_at DESC
        LIMIT $2 OFFSET $3`,
      [session.userId, pageSize, (page - 1) * pageSize],
    );
    return { total: countRows[0].n, items: rows };
  });

  return (
    <div className="p-4 md:p-6 max-w-3xl">
      <h1 className="text-xl font-bold mb-4">自分宛の依頼</h1>

      {/* Status tabs */}
      <div className="flex gap-0 border-b-2 border-gray-200 mb-4">
        <Link
          href={`/t/${code}/requests?status=pending`}
          className={`px-4 py-2 text-sm font-medium no-underline -mb-0.5 ${
            statusFilter !== 'done'
              ? 'border-b-2 border-blue-600 text-blue-600'
              : 'text-gray-500 hover:text-gray-700'
          }`}
        >
          未対応
        </Link>
        <Link
          href={`/t/${code}/requests?status=done`}
          className={`px-4 py-2 text-sm font-medium no-underline -mb-0.5 ${
            statusFilter === 'done'
              ? 'border-b-2 border-blue-600 text-blue-600'
              : 'text-gray-500 hover:text-gray-700'
          }`}
        >
          完了
        </Link>
      </div>

      {/* Cards */}
      <div className="space-y-2">
        {data.items.length === 0 && (
          <p className="text-gray-500 text-center py-8">依頼はありません</p>
        )}
        {data.items.map((item) => {
          const hasUnread =
            item.last_comment_at &&
            (!item.last_viewed_at || new Date(item.last_comment_at) > new Date(item.last_viewed_at));
          return (
            <Link
              key={item.id}
              href={`/t/${code}/requests/${item.request_id}`}
              className="block bg-white border border-gray-200 rounded-lg p-4 hover:border-gray-300 transition-colors no-underline"
            >
              <div className="flex items-start justify-between gap-2">
                <div className="flex-1 min-w-0">

                    {hasUnread && <span className="text-blue-500 text-xs">🔵</span>}
                    <span className="font-medium text-gray-900 truncate">
                      {item.title}
                    </span>
                  </div>
                  <div className="text-xs text-gray-500 mt-1">
                    {item.sender_name}
                  </div>
                </div>
                <div className="flex flex-col items-end gap-1 flex-shrink-0">
                  <StatusBadge status={item.status} isOverdue={item.is_overdue} />
                  {item.due_at && (
                    <span className={`text-xs ${item.is_overdue ? 'text-red-500 font-medium' : 'text-gray-400'}`}>
                      {new Date(item.due_at).toLocaleDateString('ja-JP')}
                    </span>
                  )}
                </div>
              </div>
            </Link>
          );
        })}
      </div>

      {/* Load more */}
      {data.total > page * pageSize && (
        <div className="text-center mt-4">
          <Link
            href={`/t/${code}/requests?status=${statusFilter}&page=${page + 1}`}
            className="text-blue-600 text-sm hover:underline"
          >
            もっと見る（残り {data.total - page * pageSize} 件）
          </Link>
        </div>
      )}
    </div>
  );
}

'use client';

import { useState, useEffect } from 'react';

type OrgTreeNode = {
  id: string;
  name: string;
  memberCount: number;
  children: OrgTreeNode[];
};

type SelectedOrg = {
  id: string;
  name: string;
  includeDescendants: boolean;
};

type Props = {
  tenantCode: string;
  selected: SelectedOrg[];
  onChange: (orgs: SelectedOrg[]) => void;
};

export function OrgTreePicker({ tenantCode, selected, onChange }: Props) {
  const [tree, setTree] = useState<OrgTreeNode[]>([]);
  const [expanded, setExpanded] = useState<Set<string>>(new Set());
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch(`/t/${tenantCode}/api/org-tree`)
      .then((r) => r.json())
      .then((data) => { setTree(data); setLoading(false); })
      .catch(() => setLoading(false));
  }, [tenantCode]);

  const selectedIds = new Set(selected.map((s) => s.id));

  function toggleExpand(id: string) {
    setExpanded((prev) => {
      const next = new Set(prev);
      next.has(id) ? next.delete(id) : next.add(id);
      return next;
    });
  }

  function toggleSelect(node: OrgTreeNode) {
    if (selectedIds.has(node.id)) {
      onChange(selected.filter((s) => s.id !== node.id));
    } else {
      onChange([...selected, { id: node.id, name: node.name, includeDescendants: true }]);
    }
  }

  function toggleDescendants(id: string) {
    onChange(selected.map((s) =>
      s.id === id ? { ...s, includeDescendants: !s.includeDescendants } : s,
    ));
  }

  function renderNode(node: OrgTreeNode, depth: number) {
    const hasChildren = node.children.length > 0;
    const isExpanded = expanded.has(node.id);
    const isSelected = selectedIds.has(node.id);

    return (
      <div key={node.id}>
        <div
          className={`flex items-center gap-1 py-1 cursor-pointer text-sm rounded px-1 ${
            isSelected ? 'bg-blue-50' : 'hover:bg-gray-50'
          }`}
          style={{ paddingLeft: depth * 20 + 4 }}
        >
          <span
            onClick={() => hasChildren && toggleExpand(node.id)}
            className="w-4 text-xs text-gray-400"
          >
            {hasChildren ? (isExpanded ? '▼' : '▶') : ''}
          </span>
          <span onClick={() => toggleSelect(node)} className="flex-1 flex items-center justify-between">
            <span>📁 {node.name} <span className="text-xs text-gray-400">({node.memberCount})</span></span>
            {isSelected && <span className="text-blue-600 text-xs">✓</span>}
          </span>
        </div>
        {hasChildren && isExpanded && node.children.map((c) => renderNode(c, depth + 1))}
      </div>
    );
  }

  if (loading) return <p className="text-sm text-gray-400">読み込み中...</p>;

  return (
    <div className="flex gap-4">
      <div className="flex-1 border border-gray-200 rounded-md bg-gray-50 p-2 max-h-60 overflow-y-auto">
        <div className="text-xs text-gray-500 font-medium mb-2">組織を選択（複数可）</div>
        {tree.map((n) => renderNode(n, 0))}
      </div>

      {selected.length > 0 && (
        <div className="w-56 flex-shrink-0">
          <div className="text-xs text-gray-500 font-medium mb-2">
            選択済み（{selected.length} 組織）
          </div>
          {selected.map((s) => (
            <div key={s.id} className="bg-blue-50 border border-blue-200 rounded-md p-2 mb-2">
              <div className="flex items-center justify-between text-sm">
                <span className="font-medium">🏢 {s.name}</span>
                <button
                  onClick={() => onChange(selected.filter((x) => x.id !== s.id))}
                  className="text-red-400 hover:text-red-600 text-xs"
                >
                  ✕
                </button>
              </div>
              <label className="flex items-center gap-1 mt-1 text-xs text-gray-600">
                <input
                  type="checkbox"
                  checked={s.includeDescendants}
                  onChange={() => toggleDescendants(s.id)}
                />
                配下含む
              </label>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}

type Props = {

  showAllTab: boolean;
};

export function TargetPicker({ tenantCode, targets, onChange, showAllTab }: Props) {
  const [tab, setTab] = useState<'org' | 'user' | 'group' | 'all'>('org');

  const orgTargets = targets.filter((t) => t.type === 'org_unit') as Array<{
    type: 'org_unit'; orgUnitId: string; includeDescendants: boolean;
  }>;
  const userTargets = targets.filter((t) => t.type === 'user') as Array<{
    type: 'user'; userId: string;
  }>;
  const hasAll = targets.some((t) => t.type === 'all');

  function setOrgTargets(orgs: Array<{ id: string; name: string; includeDescendants: boolean }>) {
    const nonOrg = targets.filter((t) => t.type !== 'org_unit');
    const newOrg: TargetSpec[] = orgs.map((o) => ({
      type: 'org_unit', orgUnitId: o.id, includeDescendants: o.includeDescendants,
    }));
    onChange([...nonOrg, ...newOrg]);
  }

  function addUserTarget(user: { id: string }) {
    if (userTargets.some((t) => t.userId === user.id)) return;
    onChange([...targets, { type: 'user', userId: user.id }]);
  }

  function removeUserTarget(userId: string) {
    onChange(targets.filter((t) => !(t.type === 'user' && t.userId === userId)));
  }

  function toggleAll() {
    if (hasAll) {
      onChange(targets.filter((t) => t.type !== 'all'));
    } else {
      onChange([...targets, { type: 'all' }]);
    }
  }

  const tabs = [
    { key: 'org' as const, label: '🏢 組織' },
    { key: 'user' as const, label: '👤 個人' },
    { key: 'group' as const, label: '👥 グループ' },
    ...(showAllTab ? [{ key: 'all' as const, label: '🌐 全社' }] : []),
  ];

  return (
    <div>
      <div className="flex gap-0 border-b-2 border-gray-200 mb-4">
        {tabs.map((t) => (
          <button
            key={t.key}
            onClick={() => setTab(t.key)}
            className={`px-4 py-2 text-sm font-medium -mb-0.5 ${
              tab === t.key
                ? 'border-b-2 border-blue-600 text-blue-600'
                : 'text-gray-500 hover:text-gray-700'
            }`}
          >
            {t.label}
          </button>
        ))}
      </div>

      {tab === 'org' && (
        <OrgTreePicker
          tenantCode={tenantCode}
          selected={orgTargets.map((o) => ({
            id: o.orgUnitId, name: '', includeDescendants: o.includeDescendants,
          }))}
          onChange={setOrgTargets}
        />
      )}

      {tab === 'user' && (

          <UserSearch

            onSelect={addUserTarget}

      {assignmentId && !isRequester && (
        <div className="bg-white border border-gray-200 rounded-lg p-4">
          <h3 className="text-sm font-semibold text-gray-700 mb-3">💬 質問・やり取り</h3>
          <div className="space-y-2 mb-3 max-h-64 overflow-y-auto">
            {myThread.length === 0 && (
              <p className="text-xs text-gray-400">まだやり取りはありません</p>
            )}
            {myThread.map((c) => {
              const isMe = c.authorUserId === currentUserId;
              return (
                <div key={c.id} className={`flex ${isMe ? 'justify-end' : 'justify-start'}`}>
                  <div className={`max-w-[70%] rounded-lg px-3 py-2 text-sm ${
                    isMe ? 'bg-blue-100 text-blue-900' : 'bg-gray-100 text-gray-800'
                  }`}>
                    <div className="text-xs font-medium mb-0.5">{c.authorName}</div>
                    <div className="whitespace-pre-wrap">{c.body}</div>
                    <div className="text-xs text-gray-400 mt-1">
                      {new Date(c.createdAt).toLocaleString('ja-JP')}
                    </div>
                  </div>
                </div>
              );
            })}
          </div>
          <div className="flex gap-2">
            <input
              type="text"
              value={newBody}
              onChange={(e) => setNewBody(e.target.value)}
              placeholder="メッセージを入力..."
              className="flex-1 border border-gray-300 rounded-md px-3 py-1.5 text-sm"
              onKeyDown={(e) => {
                if (e.key === 'Enter' && !e.shiftKey && newBody.trim()) {
                  e.preventDefault();
                  postComment(newBody, assignmentId);
                }
              }}
            />
            <button
              onClick={() => postComment(newBody, assignmentId)}
              disabled={sending || !newBody.trim()}
              className="px-4 py-1.5 bg-blue-600 text-white rounded-md text-sm disabled:opacity-50"
            >
              送信
            </button>
          </div>
        </div>
      )}

      {/* All threads (requester view) */}
      {isRequester && allThreads && Object.keys(allThreads).length > 0 && (
        <div className="bg-white border border-gray-200 rounded-lg p-4 mt-4">
          <h3 className="text-sm font-semibold text-gray-700 mb-3">💬 個別やり取り一覧</h3>
          {Object.entries(allThreads).map(([asgId, thread]) => (
            <details key={asgId} className="mb-2 border border-gray-100 rounded-md">
              <summary className="px-3 py-2 text-sm cursor-pointer hover:bg-gray-50">
                {thread[0]?.authorName ?? asgId}（{thread.length} 件）
              </summary>
              <div className="px-3 py-2 space-y-2">
                {thread.map((c) => (
                  <div key={c.id} className="text-sm">
                    <span className="font-medium">{c.authorName}:</span>{' '}
                    <span className="text-gray-700">{c.body}</span>
                    <span className="text-xs text-gray-400 ml-2">
                      {new Date(c.createdAt).toLocaleString('ja-JP')}
                    </span>
                  </div>
                ))}
                {/* Inline reply */}
                <ReplyInput
                  onSend={(body) => postComment(body, asgId)}
                  disabled={sending}
                />
              </div>
            </details>
          ))}
        </div>
      )}
    </div>
  );
}

function ReplyInput({ onSend, disabled }: { onSend: (body: string) => void; disabled: boolean }) {
  const [body, setBody] = useState('');
  return (
    <div className="flex gap-2 mt-2">
      <input
        type="text"
        value={body}
        onChange={(e) => setBody(e.target.value)}
        placeholder="返信..."
        className="flex-1 border border-gray-300 rounded-md px-2 py-1 text-sm"
      />
      <button
        onClick={() => { onSend(body); setBody(''); }}
        disabled={disabled || !body.trim()}
        className="px-3 py-1 bg-blue-600 text-white rounded-md text-xs disabled:opacity-50"
      >
        返信
      </button>
    </div>
  );
}
